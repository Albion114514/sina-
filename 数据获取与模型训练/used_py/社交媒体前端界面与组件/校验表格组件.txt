<!-- ProcessedTable.vue -->
<template>
  <div class="processed-table">
    <el-button 
      type="primary" 
      icon="el-icon-refresh" 
      @click="$emit('refresh')"
      :loading="loading"
    >
      刷新
    </el-button>
    
    <el-table :data="data" :loading="loading" style="width: 100%; margin-top: 20px">
      <el-table-column 
        prop="text" 
        label="文本内容" 
        min-width="300"
        show-overflow-tooltip
      >
        <template slot-scope="scope">
          {{ scope.row.text }}
        </template>
      </el-table-column>
      
      <el-table-column 
        prop="nature" 
        label="性质" 
        width="100"
        v-if="type === 'processed'"
      >
        <template slot-scope="scope">
          <el-tag 
            :type="scope.row.nature === 'rumor' ? 'danger' : 'success'"
            size="small"
          >
            {{ scope.row.nature === 'rumor' ? '谣言' : '非谣言' }}
          </el-tag>
        </template>
      </el-table-column>
      
      <el-table-column prop="source_platform" label="来源" width="100"></el-table-column>
      
      <el-table-column 
        prop="rumor_probability" 
        label="谣言概率" 
        width="120"
        v-if="type === 'unprocessed'"
      >
        <template slot-scope="scope">
          <span :class="getProbabilityClass(scope.row.rumor_probability)">
            {{ scope.row.rumor_probability }}%
          </span>
        </template>
      </el-table-column>
      
      <el-table-column 
        prop="process_time" 
        label="处理时间" 
        width="150"
        v-if="type === 'processed'"
      >
        <template slot-scope="scope">
          {{ formatTime(scope.row.process_time) }}
        </template>
      </el-table-column>
      
      <el-table-column 
        prop="timestamp" 
        label="发布时间" 
        width="150"
        v-if="type === 'unprocessed'"
      >
        <template slot-scope="scope">
          {{ formatTime(scope.row.timestamp) }}
        </template>
      </el-table-column>
      
      <el-table-column label="操作" width="120">
        <template slot-scope="scope">
          <el-button 
            v-if="type === 'unprocessed'"
            size="mini" 
            type="primary" 
            @click="$emit('process', scope.row)"
          >
            处理
          </el-button>
          
          <el-button 
            v-if="type === 'processed'"
            size="mini" 
            type="warning" 
            @click="$emit('modify', scope.row)"
          >
            修改
          </el-button>
        </template>
      </el-table-column>
    </el-table>
    
    <!-- 分页 -->
    <div class="pagination" v-if="data.length > 0">
      <el-pagination
        background
        layout="prev, pager, next"
        :total="total"
        :page-size="20"
        @current-change="handlePageChange"
      >
      </el-pagination>
    </div>
  </div>
</template>

<script>
export default {
  name: 'ProcessedTable',
  props: {
    data: {
      type: Array,
      default: () => []
    },
    loading: {
      type: Boolean,
      default: false
    },
    type: {
      type: String, // 'processed' or 'unprocessed'
      required: true
    },
    total: {
      type: Number,
      default: 0
    }
  },
  methods: {
    getProbabilityClass(probability) {
      if (probability >= 90) return 'high-risk'
      if (probability >= 60) return 'medium-risk'
      return 'low-risk'
    },
    
    formatTime(timestamp) {
      if (!timestamp) return ''
      return new Date(timestamp).toLocaleString()
    },
    
    handlePageChange(page) {
      this.$emit('page-change', page)
    }
  }
}
</script>

<style scoped>
.high-risk {
  color: #ff4d4f;
  font-weight: bold;
}

.medium-risk {
  color: #faad14;
  font-weight: bold;
}

.low-risk {
  color: #52c41a;
}

.pagination {
  margin-top: 20px;
  text-align: center;
}
</style>
